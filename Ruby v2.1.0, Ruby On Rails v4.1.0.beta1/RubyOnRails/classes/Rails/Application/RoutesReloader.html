def clear!
  route_sets.each do |routes|
    routes.disable_clear_and_finalize = true
    routes.clear!
  end

def updater
  @updater ||= begin
    updater = ActiveSupport::FileUpdateChecker.new(paths) { reload! }
    updater.execute
    updater
  end
