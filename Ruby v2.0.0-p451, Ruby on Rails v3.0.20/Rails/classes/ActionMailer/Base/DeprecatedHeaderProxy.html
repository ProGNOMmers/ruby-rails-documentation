def []=(key, value)
  unless value.is_a?(::String)
    ::ActiveSupport::Deprecation.warn("Using a non-String object for a header's value is deprecated. "              "You specified #{value.inspect} (a #{value.class}) for #{key}", caller)

    value = value.to_s
  end

  @message[key] = value
end
