def validate_each(record, attribute, value)
  included = if options[:in].is_a?(Range)
    options[:in].cover?(value)
  else
    options[:in].include?(value)
  end

  unless included
    record.errors.add(attribute, :inclusion, options.except(:in).merge!(:value => value))
  end
end
