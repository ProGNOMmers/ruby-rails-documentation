def all_application_helpers
  helpers = []
  Array.wrap(helpers_path).each do |path|
    extract  = /^#{Regexp.quote(path.to_s)}\/?(.*)_helper.rb$/
    helpers += Dir["#{path}/**/*_helper.rb"].map { |file| file.sub(extract, '\1') }
  end
  helpers.sort!
  helpers.uniq!
  helpers
end
