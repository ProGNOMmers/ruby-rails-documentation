def filter_by(constraint)
  all = []
  all_paths.each do |path|
    if path.send(constraint)
      paths  = path.paths
      paths -= path.children.values.map { |p| p.send(constraint) ? [] : p.paths }.flatten
      all.concat(paths)
    end
  end
  all.uniq!
  all.reject! { |p| !File.exists?(p) }
  all
end
